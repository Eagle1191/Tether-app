CREATE TABLE USER
(
  Name_ID VARCHAR(255) NOT NULL,
  SteamID INT NOT NULL,
  Region VARCHAR(255) NOT NULL,
  PRIMARY KEY (Name_ID)
);

CREATE TABLE League
(
  League_ID INT NOT NULL,
  Region VARCHAR(255) NOT NULL,
  Skill_Level VARCHAR(255) NOT NULL,
  Members VARCHAR(255) NOT NULL,
  Name_ID VARCHAR(255) NOT NULL,
  PRIMARY KEY (League_ID),
  FOREIGN KEY (Name_ID) REFERENCES USER(Name_ID)
);

CREATE TABLE PROFILE
(
  Region VARCHAR(255) NOT NULL,
  Winrate NUMERIC NOT NULL,
  Average_GPM INT NOT NULL,
  League_Rank VARCHAR(255) NOT NULL,
  SteamID INT NOT NULL,
  Name_ID VARCHAR(255) NOT NULL,
  PRIMARY KEY (SteamID),
  FOREIGN KEY (Name_ID) REFERENCES USER(Name_ID)
);

CREATE TABLE RECENT_MATCHES
(
  Skill_Level VARCHAR(255) NOT NULL,
  Date DATE NOT NULL,
  SteamID INT NOT NULL,
  Hero_ID INT NOT NULL,
  Result_WL VARCHAR(255) NOT NULL,
  KDA FLOAT NOT NULL,
  GPM FLOAT NOT NULL,
  Match_ID INT NOT NULL,
  PRIMARY KEY (Match_ID),
  FOREIGN KEY (SteamID) REFERENCES PROFILE(SteamID),
  UNIQUE (Hero_ID)
);
